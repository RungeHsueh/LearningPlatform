    <script>
        lucide.createIcons();

        // 偵測是否為 iOS 設備 (含新版 iPad)
        function isIOS() {
            return [
                'iPad Simulator',
                'iPhone Simulator',
                'iPod Simulator',
                'iPad',
                'iPhone',
                'iPod'
            ].includes(navigator.platform)
            || (navigator.userAgent.includes("Mac") && "ontouchend" in document);
        }

        if (isIOS()) {
            document.getElementById('ios-mic-tip').classList.remove('hidden');
        }

        let isSidebarOpen = false;
        const sidebar = document.getElementById('sidebar');
        const overlay = document.getElementById('sidebar-overlay');

        function toggleSidebar() {
            const isMobile = window.innerWidth < 768;
            if (isMobile) { 
                isSidebarOpen = !isSidebarOpen;
                if (isSidebarOpen) {
                    sidebar.classList.remove('-translate-x-full');
                    sidebar.classList.add('translate-x-0');
                    overlay.classList.remove('opacity-0', 'pointer-events-none');
                    overlay.classList.add('opacity-100');
                } else {
                    sidebar.classList.remove('translate-x-0');
                    sidebar.classList.add('-translate-x-full');
                    overlay.classList.remove('opacity-100');
                    overlay.classList.add('opacity-0', 'pointer-events-none');
                }
            }
        }

        function toggleMenu(menuId, arrowId) {
            const menu = document.getElementById(menuId);
            const arrow = document.getElementById(arrowId);
            
            document.querySelectorAll('.submenu').forEach(el => {
                if(el.id !== menuId) { el.classList.remove('open'); }
            });
            document.querySelectorAll('.arrow-icon').forEach(el => {
                if(el.id !== arrowId) { el.classList.remove('rotate'); }
            });

            menu.classList.toggle('open');
            arrow.classList.toggle('rotate');
        }

        function loadContent(event, url, title) {
            event.preventDefault();
            const titleText = document.getElementById('title-text');
            const welcomeScreen = document.getElementById('welcome-screen');
            const courseGuide = document.getElementById('course-guide');
            const selectedCourseTitle = document.getElementById('selected-course-title');
            const externalLink = document.getElementById('external-link');
            const hintMessage = document.getElementById('hint-message');

            welcomeScreen.classList.add('hidden');
            courseGuide.classList.remove('hidden'); 
            courseGuide.style.display = 'block'; 

            titleText.innerText = title;
            selectedCourseTitle.innerText = title;
            externalLink.href = url;
            externalLink.classList.remove('hidden');
            externalLink.classList.add('inline-flex');
            hintMessage.classList.remove('opacity-0');

            if (window.innerWidth < 768 && isSidebarOpen) {
                toggleSidebar();
            }
        }
        
        document.addEventListener('DOMContentLoaded', () => {
            lucide.createIcons();
            const hintMessage = document.getElementById('hint-message');
            hintMessage.classList.add('opacity-0');
        });

        window.addEventListener('resize', () => {
            if (window.innerWidth >= 768) {
                sidebar.classList.remove('-translate-x-full', 'translate-x-0');
                overlay.classList.remove('opacity-100');
                overlay.classList.add('opacity-0', 'pointer-events-none');
                isSidebarOpen = false;
            }
        });
    </script>
